void setup() {
  Serial.begin(9600);
  while(!Serial);
}

void loop() {
  int i, j;
  for(i = 0 ; i <= 5 ; i++)
  {
    for(j = i ; j < 5 ; j++)
    {
      Serial.print(" ");
    }

    for(j = 10 - 1 - i * 2 ; j < 10 ; j++)
    {
      Serial.print("*");
    }
    Serial.println();
  }

  delay(10000);
}
